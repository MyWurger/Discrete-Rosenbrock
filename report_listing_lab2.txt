#include <vector>

namespace lab2

// ----------------------------------------------------------------------------
// ТИП Vector - ВЕКТОР КООРДИНАТ ТОЧКИ В ПРОСТРАНСТВЕ ПЕРЕМЕННЫХ
// ----------------------------------------------------------------------------
// Используется для:
// - хранения x = (x1, x2, ..., xn);
// - передачи точки в Evaluate(...);
// - хранения результатов метода и стационарной точки.
// ----------------------------------------------------------------------------
typedef std::vector<double> Vector;

// ----------------------------------------------------------------------------
// ИНТЕРФЕЙС IObjectiveFunction - АБСТРАКТНАЯ ЦЕЛЕВАЯ ФУНКЦИЯ
// ----------------------------------------------------------------------------
// Через этот интерфейс метод Розенброка работает с функцией независимо от того,
// как она была создана: вручную, через muparser или иным способом.
// ----------------------------------------------------------------------------
class IObjectiveFunction
{
public:
    // Виртуальный деструктор нужен для корректного удаления через указатель на базу.
    virtual ~IObjectiveFunction()
    {
    }

    // Возвращает размерность пространства, в котором задана функция.
    virtual std::size_t Dimension() const = 0;

    // Вычисляет значение функции в точке x.
    // Принимает:
    // - x: вектор координат точки;
    // - value: ссылка, куда записывается значение функции при успехе;
    // - error: строка для текста ошибки при неудачном вычислении.
    // Возвращает:
    // - true, если значение успешно вычислено;
    // - false, если вычисление выполнить не удалось.
    virtual bool Evaluate(const Vector& x, double& value, std::string& error) const = 0;

    // Возвращает имя функции для таблиц, GUI и отчетов.
    virtual const std::string& Name() const = 0;
};

}  // namespace lab2

// ИНТЕРФЕЙС IObjectiveParser - АБСТРАКТНЫЙ ПАРСЕР СТРОКОВОЙ ФОРМУЛЫ
// ----------------------------------------------------------------------------
// Реализация этого интерфейса должна:
// - принять текст выражения;
// - проверить его корректность;
// - создать объект IObjectiveFunction, пригодный для дальнейших вычислений.

class IObjectiveParser
{
public:
    // Виртуальный деструктор нужен для корректного удаления через указатель на базу.
    virtual ~IObjectiveParser()
    {

    // Разбирает текстовое выражение и создает объект функции.
    // Принимает:
    // - expression: текст математической формулы;

    // - name: имя функции, которое будет сохранено внутри создаваемого объекта;

    // - true, если парсинг завершился успешно;
    // - false, если выражение некорректно или объект создать не удалось.

                       std::size_t dimension,
                       const std::string& name,
                       IObjectiveFunction*& output,
                       std::string& error) const = 0;
};

// ===== FILE: include/lab2/RosenbrockDiscreteMethod.h =====
// ============================================================================
// ФАЙЛ ROSENBROCKDISCRETEMETHOD.H - КОНТРАКТ ДИСКРЕТНОГО МЕТОДА РОЗЕНБРОКА
// ============================================================================
// Назначение файла:
// 1) Объявить входные параметры метода Розенброка с дискретным шагом.
// 2) Описать структуру результата и формат полной трассировки шагов.
// 3) Объявить интерфейс оптимизатора и конкретную реализацию метода.
// ============================================================================

// Защита от повторного включения заголовочного файла.
#pragma once

// Подключаем интерфейс целевой функции и тип Vector.
#include "ObjectiveFunction.h"

// std::size_t нужен для размерностей, индексов и счетчиков.
#include <cstddef>
// std::string нужен для текстов ошибок, сообщений и строкового представления базиса.
#include <string>
// std::vector нужен для хранения точек, направлений и полной трассировки.
#include <vector>

namespace lab2
{

// ----------------------------------------------------------------------------
// СТРУКТУРА RosenbrockDiscreteOptions - ПАРАМЕТРЫ ДИСКРЕТНОГО МЕТОДА
// ----------------------------------------------------------------------------
// Содержит все численные параметры, которые задаются перед запуском метода:
// - epsilon;
// - alpha и beta;
// - начальные длины шагов Delta_j;
// - предохранительный лимит числа внешних итераций.
// ----------------------------------------------------------------------------
struct RosenbrockDiscreteOptions
{
    double epsilon;                    // Точность epsilon для критериев остановки метода.
    double alpha;                      // Коэффициент растяжения alpha, должен быть больше 1.
    double beta;                       // Коэффициент сжатия beta, должен лежать в интервале (-1, 0).
    Vector initialStepMagnitudes;      // Начальные длины шагов Delta_j вдоль каждого направления.
    std::size_t maxIterations;         // Предохранительный лимит числа внешних итераций k.

    // Конструктор по умолчанию заполняет параметры стандартными значениями проекта.
    RosenbrockDiscreteOptions();
};

// ----------------------------------------------------------------------------
// СТРУКТУРА OptimizationRequest - ПОЛНЫЙ ЗАПРОС НА ОПТИМИЗАЦИЮ
// ----------------------------------------------------------------------------
// Объединяет:
// - целевую функцию;
// - стартовую точку x1;
// - набор параметров метода.
// ----------------------------------------------------------------------------
struct OptimizationRequest
{
    const IObjectiveFunction* function;   // Целевая функция, которую нужно оптимизировать.
    Vector initialPoint;                  // Начальная точка x1 метода.
    RosenbrockDiscreteOptions options;    // Параметры дискретного метода Розенброка.

    // Конструктор по умолчанию создает пустой/базовый запрос.
    OptimizationRequest();
};

// ----------------------------------------------------------------------------
// СТРУКТУРА OptimizationResult - РЕЗУЛЬТАТ РАБОТЫ МЕТОДА
// ----------------------------------------------------------------------------
// Хранит:
// - признак сходимости;
// - итоговую точку и значение функции;
// - статистику по итерациям и вычислениям;
// - полную трассировку шагов метода для таблицы и графиков.
// ----------------------------------------------------------------------------
struct OptimizationResult
{
    // ------------------------------------------------------------------------
    // СТРУКТУРА TraceRow - ОДНА СТРОКА ТРАССИРОВКИ МЕТОДА
    // ------------------------------------------------------------------------
    // Каждая строка соответствует одному конкретному шагу j внутри внешней
    // итерации k и хранит все значения, которые нужны таблице и графикам.
    // ------------------------------------------------------------------------
    struct TraceRow
    {
        std::size_t k;                            // Номер внешней итерации метода.
        std::size_t j;                            // Номер шага по текущему направлению внутри итерации.

        Vector xk;                                // Базисная точка x_k текущей внешней итерации.
        double fxk;                               // Значение функции в точке x_k.

        Vector yi;                                // Текущая базисная точка y_j перед пробным шагом.
        double fyi;                               // Значение функции в точке y_j.

        double delta;                             // Текущее значение Delta_j для рассматриваемого направления.
        Vector direction;                         // Текущее направление s_j.

        Vector trialPoint;                        // Пробная точка y_j + Delta_j * s_j.
        double fTrial;                            // Значение функции в пробной точке.

        bool successfulStep;                      // Признак удачности пробного шага.
        bool rollback;                            // Признак отката после серии неудачных шагов.
        bool directionChanged;                    // Признак перестройки базиса направлений.
        std::string newBasis;                     // Строковое представление нового базиса для таблицы.
        std::vector<Vector> newBasisDirections;   // Новый базис направлений, построенный на этом шаге.
        Vector newBasisOrigin;                    // Точка, из которой начинается новый базис.
        std::vector<Vector> basisDirections;      // Базис направлений, действующий на текущем шаге.
        Vector basisOrigin;                       // Точка привязки действующего базиса на текущем шаге.

        // Конструктор по умолчанию создает пустую строку трассировки.
        TraceRow();
    };

    bool converged;                    // Признак того, что метод завершился по критерию сходимости.
    std::string message;               // Итоговое текстовое сообщение о результате работы метода.

    Vector optimumX;                   // Найденная оптимальная точка x*.
    double optimumValue;               // Значение функции в найденной точке x*.
    Vector finalStepMagnitudes;        // Итоговый набор длин шагов Delta_j в момент завершения метода.

    std::size_t iterations;            // Число внешних итераций k, выполненных методом.
    std::size_t functionEvaluations;   // Общее число вычислений целевой функции.

    std::vector<TraceRow> trace;       // Полная трассировка шагов метода для таблицы и графиков.

    // Конструктор по умолчанию создает пустой результат.
    OptimizationResult();
};

// ----------------------------------------------------------------------------
// ИНТЕРФЕЙС IOptimizer - ОБЩИЙ ИНТЕРФЕЙС ОПТИМИЗАТОРА
// ----------------------------------------------------------------------------
// Позволяет запускать конкретный алгоритм через единый метод Optimize(...).
// ----------------------------------------------------------------------------
class IOptimizer
{
public:
    // Виртуальный деструктор нужен для корректной работы через указатель на базу.
    virtual ~IOptimizer()
    {
    }

    // Запускает процесс оптимизации по заданному запросу.
    // Принимает:
    // - request: полный набор входных данных;
    // - result: структура, в которую будет записан итог работы;
    // - error: строка для текста ошибки, если метод не удалось выполнить.
    // Возвращает:
    // - true, если выполнение завершилось корректно;
    // - false, если произошла ошибка валидации или вычисления.
    virtual bool Optimize(const OptimizationRequest& request,
                          OptimizationResult& result,
                          std::string& error) const = 0;
};

// ----------------------------------------------------------------------------
// КЛАСС RosenbrockDiscreteOptimizer - КОНКРЕТНАЯ РЕАЛИЗАЦИЯ МЕТОДА
// ----------------------------------------------------------------------------
// Реализует дискретный метод Розенброка в соответствии с алгоритмом из
// методички: проход по направлениям, изменение Delta_j, перестройка базиса
// и проверка критериев остановки.
// ----------------------------------------------------------------------------
class RosenbrockDiscreteOptimizer : public IOptimizer
{
public:
    // Основной запуск дискретного метода Розенброка.
    // Принимает:
    // - request: полный набор входных данных задачи оптимизации.
    // Записывает:
    // - result: итог работы метода, включая точку x*, статистику и трассировку;
    // - error: текст ошибки, если выполнить метод корректно не удалось.
    // Возвращает:
    // - true, если метод завершился без внутренних ошибок;
    // - false, если произошла ошибка валидации входа или вычисления функции.
    bool Optimize(const OptimizationRequest& request,
                  OptimizationResult& result,
                  std::string& error) const;

private:
    // Проверяет корректность численных параметров метода.
    // Принимает:
    // - options: набор численных параметров метода.
    // Записывает:
    // - error: текст ошибки, если найдено некорректное значение параметра.
    // Возвращает:
    // - true, если параметры согласованы с требованиями метода;
    // - false, если хотя бы один параметр некорректен.
    static bool ValidateOptions(const RosenbrockDiscreteOptions& options, std::string& error);

    // Формирует полный вектор начальных длин шагов Delta_j по размерности задачи.
    // Принимает:
    // - options: параметры метода, включая пользовательский ввод по Delta_j;
    // - dimension: размерность текущей задачи.
    // Записывает:
    // - steps: итоговый вектор начальных шагов по всем направлениям;
    // - error: текст ошибки, если построить корректный вектор шагов не удалось.
    // Возвращает:
    // - true, если вектор шагов успешно подготовлен;
    // - false, если размерность или входные значения некорректны.
    static bool BuildInitialStepMagnitudes(const RosenbrockDiscreteOptions& options,
                                           std::size_t dimension,
                                           Vector& steps,
                                           std::string& error);

    // Безопасно вычисляет значение функции и увеличивает счетчик вычислений.
    // Принимает:
    // - function: указатель на целевую функцию;
    // - x: точка, в которой нужно вычислить значение.
    // Записывает:
    // - value: вычисленное значение функции при успехе;
    // - evalCounter: увеличивается на единицу при каждом фактическом вызове Evaluate(...);
    // - error: текст ошибки, если вычисление завершилось неудачно.
    // Возвращает:
    // - true, если значение функции успешно получено;
    // - false, если функция невалидна или вычисление завершилось ошибкой.
    static bool EvaluateObjective(const IObjectiveFunction* function,
                                  const Vector& x,
                                  double& value,
                                  std::size_t& evalCounter,
                                  std::string& error);

    // Скалярное произведение двух векторов.
    // Принимает:
    // - a, b: два вектора одинаковой размерности.
    // Возвращает:
    // - число a * b в смысле скалярного произведения.
    static double Dot(const Vector& a, const Vector& b);

    // Евклидова норма вектора.
    // Принимает:
    // - values: произвольный вектор координат.
    // Возвращает:
    // - длину вектора в евклидовой норме.
    static double Norm(const Vector& values);

    // Разность двух векторов a - b.
    // Принимает:
    // - a, b: два вектора одинаковой размерности.
    // Возвращает:
    // - новый вектор, равный a - b.
    static Vector Subtract(const Vector& a, const Vector& b);

    // Добавляет к target вектор scale * direction.
    // Принимает:
    // - scale: числовой коэффициент;
    // - direction: направление, которое нужно прибавить с масштабированием.
    // Изменяет:
    // - target: после вызова становится равным target + scale * direction.
    static void AddScaled(Vector& target, double scale, const Vector& direction);

    // Проверяет, что все |Delta_j| стали меньше epsilon.
    // Принимает:
    // - deltas: текущий набор длин шагов Delta_j;
    // - epsilon: требуемый порог точности.
    // Возвращает:
    // - true, если для всех направлений выполняется |Delta_j| < epsilon;
    // - false, если хотя бы один шаг еще слишком велик.
    static bool AllStepMagnitudesBelowEpsilon(const Vector& deltas, double epsilon);

    // Строит начальный базис единичных направлений e_j.
    // Принимает:
    // - dimension: размерность пространства переменных.
    // Возвращает:
    // - вектор направлений {e_1, e_2, ..., e_n}.
    static std::vector<Vector> IdentityDirections(std::size_t dimension);

    // Ортогонализует кандидатное направление с резервным сценарием при вырождении.
    // Принимает:
    // - candidate: направление-кандидат для включения в новый базис;
    // - basis: уже построенная часть нового базиса;
    // - oldDirections: предыдущий базис направлений;
    // - index: номер направления, которое сейчас строится.
    // Записывает:
    // - outDirection: готовое ортогонализованное направление;
    // - error: текст ошибки, если получить направление не удалось.
    // Возвращает:
    // - true, если направление успешно построено;
    // - false, если даже резервный сценарий не помог получить корректный вектор.
    static bool OrthogonalizeWithFallback(const Vector& candidate,
                                          const std::vector<Vector>& basis,
                                          const std::vector<Vector>& oldDirections,
                                          std::size_t index,
                                          Vector& outDirection,
                                          std::string& error);

    // Строит новый базис направлений по накопленным коэффициентам lambda_j.
    // Принимает:
    // - oldDirections: предыдущий базис направлений s_j;
    // - lambdas: коэффициенты разложения смещения x_{k+1} - x_k по старому базису.
    // Записывает:
    // - newDirections: новый базис направлений после перестройки;
    // - error: текст ошибки, если построить новый базис не удалось.
    // Возвращает:
    // - true, если новый базис успешно построен;
    // - false, если при перестройке возникло вырождение или иная ошибка.
    static bool BuildNewDirections(const std::vector<Vector>& oldDirections,
                                   const Vector& lambdas,
                                   std::vector<Vector>& newDirections,
                                   std::string& error);
};

// ============================================================================
// ФУНКЦИЯ ToStdString - ПЕРЕВОД СТРОКИ MUPARSER В std::string
// ============================================================================
// muparser использует собственный строковый тип mu::string_type.
// В остальном проекте мы работаем с обычным std::string, поэтому нужен
// аккуратный переход между этими типами.
// Принимает:
// - text: строка muparser.
// Возвращает:
// - обычный std::string с тем же содержимым.
// ============================================================================
static std::string ToStdString(const mu::string_type& text)
{
    // Конструируем std::string по диапазону символов mu::string_type.
    // Такой способ не зависит от внутренней реализации mu::string_type и
    // работает как для narrow-, так и для compatible wide-типов строки.
    return std::string(text.begin(), text.end());
}

// ============================================================================
// ФУНКЦИЯ BuildMuParserError - ПОСТРОЕНИЕ ПОНЯТНОГО ТЕКСТА ОШИБКИ MUPARSER
// ============================================================================
// Собирает единое человеко-читаемое сообщение из исключения muparser:
// 1) основной текст ошибки;
// 2) позиция в строке, если parser смог ее определить;
// 3) проблемный токен, если он доступен.
// Принимает:
// - parserError: исключение, полученное от muparser.
// Возвращает:
// - готовую строку ошибки, пригодную для GUI и логики валидации.
// ============================================================================
static std::string BuildMuParserError(const mu::Parser::exception_type& parserError)
{
    // Берем базовое текстовое сообщение из muparser и приводим его к std::string.
    std::string message = ToStdString(parserError.GetMsg());

    // Получаем позицию символа, на котором parser обнаружил проблему.
    // Значение >= 0 означает, что позиция действительно известна.
    const int position = parserError.GetPos();
    if (position >= 0)
    {
        // Дописываем позицию к сообщению, чтобы пользователь видел место ошибки в формуле.
        message += " (позиция: " + std::to_string(position) + ")";
    }

    // Получаем токен (фрагмент текста), вызвавший ошибку, если muparser его сохранил.
    const std::string token = ToStdString(parserError.GetToken());
    if (!token.empty())
    {
        // Если токен известен, добавляем и его тоже.
        message += ", токен: " + token;
    }

    // Возвращаем полностью собранный текст.
    return message;
}

// ============================================================================
// КЛАСС ParsedMuParserFunction - КОНКРЕТНАЯ МНОГОМЕРНАЯ ФУНКЦИЯ НА БАЗЕ MUPARSER
// ============================================================================
// Этот класс:
// - хранит имя и текст выражения;
// - создает и настраивает mu::Parser;
// - привязывает x1, x2, x3 к элементам массива variables_;
// - реализует интерфейс IObjectiveFunction.
// ============================================================================
class ParsedMuParserFunction : public IObjectiveFunction
{
public:
    // Конструктор по умолчанию.
    // Сразу выставляем dimension_ в 0, чтобы можно было распознавать
    // неинициализированное состояние объекта.
    ParsedMuParserFunction() : dimension_(0)
    {
    }

    // ------------------------------------------------------------------------
    // МЕТОД Initialize - ПОДГОТОВКА ПАРСЕРА И ПРОВЕРКА ФОРМУЛЫ
    // ------------------------------------------------------------------------
    // Шаги:
    // 1) валидируем имя, выражение и размерность;
    // 2) сохраняем имя/текст/размерность внутрь объекта;
    // 3) создаем массив переменных нужной длины;
    // 4) регистрируем x1, x2, ..., xn и короткое имя x;
    // 5) регистрируем константы pi и e;
    // 6) передаем выражение parser-у;
    // 7) делаем пробный Eval(), чтобы поймать синтаксические ошибки заранее.
    // Принимает:
    // - name: имя функции для таблиц и GUI;
    // - expression: текст математической формулы;
    // - dimension: размерность пространства переменных;
    // Записывает:
    // - error: текст ошибки, если инициализация завершится неудачно.
    // Возвращает:
    // - true, если функция успешно инициализирована;
    // - false, если входные данные некорректны или parser не смог принять формулу.
    // ------------------------------------------------------------------------
    bool Initialize(const std::string& name,
                    const std::string& expression,
                    std::size_t dimension,
                    std::string& error)
    {
        // Сначала очищаем предыдущий текст ошибки, чтобы не держать устаревшее сообщение.
        error.clear();

        // Пустое имя не допускаем: оно используется в таблицах, графиках и итогах GUI.
        if (name.empty())
        {
            error = "Имя функции не должно быть пустым";
            return false;
        }

        // Пустую формулу передавать в parser бессмысленно.
        if (expression.empty())
        {
            error = "Выражение не должно быть пустым";
            return false;
        }

        // Размерность задачи должна быть положительной.
        if (dimension == 0)
        {
            error = "Размерность должна быть положительной";
            return false;
        }

        // Сохраняем имя функции внутрь объекта для дальнейшего возврата через Name().
        name_ = name;
        // Сохраняем оригинальный текст выражения.
        expression_ = expression;
        // Сохраняем размерность.
        dimension_ = dimension;
        // Подготавливаем массив переменных нужной длины и инициализируем его нулями.
        // assign(...) удобен тем, что сразу задает и размер, и начальное значение элементов.
        variables_.assign(dimension_, 0.0);

        try
        {
            // Регистрируем имена x1, x2, ..., xn.
            // Каждый DefineVar связывает строковое имя переменной в формуле
            // с конкретным адресом double внутри variables_.
            std::size_t i = 0;
            for (i = 0; i < dimension_; ++i)
            {
                parser_.DefineVar("x" + std::to_string(i + 1), &variables_[i]);
            }

            // Дополнительно регистрируем короткое имя x как alias для x1.
            // Это полезно для одномерных выражений и делает parser совместимым
            // со старым форматом вида "x^2 + 1".
            parser_.DefineVar("x", &variables_[0]);

            // Регистрируем математическую константу pi.
            parser_.DefineConst("pi", MATH_PI_VALUE);
            // Регистрируем математическую константу e.
            parser_.DefineConst("e", MATH_E_VALUE);

            // Передаем полный текст выражения в muparser.
            // Parser не вычисляет выражение окончательно на этом шаге, а только
            // подготавливает внутреннее представление.
            parser_.SetExpr(expression_);

            // Делаем пробное вычисление сразу.
            // (void) нужен только чтобы явно показать: возвращаемое значение здесь
            // нам не важно, важен сам факт успешного Eval() без исключения.
            (void)parser_.Eval();
        }
        catch (const mu::Parser::exception_type& ex)
        {
            // Если muparser выдал исключение, конвертируем его в понятное сообщение.
            error = "Ошибка разбора выражения: " + BuildMuParserError(ex);
            return false;
        }

        // Если все шаги прошли успешно, объект готов к вычислениям.
        return true;
    }

    // Возвращает размерность функции.
    std::size_t Dimension() const
    {
        return dimension_;
    }

    // ------------------------------------------------------------------------
    // МЕТОД Evaluate - ВЫЧИСЛЕНИЕ ФУНКЦИИ В МНОГОМЕРНОЙ ТОЧКЕ
    // ------------------------------------------------------------------------
    // Принимает:
    // - x: вектор координат точки, в которой нужно вычислить функцию.
    // Записывает:
    // - value: значение функции при успешном вычислении;
    // - error: текст ошибки, если вычисление завершится неудачно.
    // Возвращает:
    // - true, если значение успешно вычислено;
    // - false, если объект не инициализирован, размерность точки неверна,
    //   координаты некорректны или muparser выдал ошибку.
    // ------------------------------------------------------------------------
    bool Evaluate(const Vector& x, double& value, std::string& error) const
    {
        // Сбрасываем старый текст ошибки перед новым вычислением.
        error.clear();

        // Если размерность равна нулю, Initialize(...) не вызывался или завершился неуспешно.
        if (dimension_ == 0)
        {
            error = "Функция не инициализирована";
            return false;
        }

        // Размерность точки обязана совпадать с размерностью функции.
        if (x.size() != dimension_)
        {
            error = "Размерность точки не совпадает с размерностью функции";
            return false;
        }

        // Переписываем все координаты входной точки в массив variables_,
        // к элементам которого уже привязан parser.
        std::size_t i = 0;
        for (i = 0; i < dimension_; ++i)
        {
            // Каждая координата обязана быть конечным числом.
            // std::isfinite защищает от NaN, +Inf и -Inf, которые ломают численные методы.
            if (!std::isfinite(x[i]))
            {
                error = "Точка должна содержать только конечные значения";
                return false;
            }

            // variables_ объявлен как mutable, потому что формально метод const,
            // но для вычисления нужно временно подставить новые значения переменных.
            variables_[i] = x[i];
        }

        try
        {
            // После обновления variables_ parser использует уже новые значения x1..xn.
            value = parser_.Eval();
        }
        catch (const mu::Parser::exception_type& ex)
        {
            // Любую ошибку вычисления переводим в человеко-читаемое сообщение.
            error = "Ошибка вычисления выражения: " + BuildMuParserError(ex);
            return false;
        }

        // Дополнительно проверяем, что итог вычисления конечный.
        // Это отлавливает случаи вроде деления на ноль, переполнения или недопустимых операций.
        if (!std::isfinite(value))
        {
            error = "Вычисление выражения дало неконечное значение";
            return false;
        }

        // Значение успешно вычислено.
        return true;
    }

    // Возвращает сохраненное имя функции.
    const std::string& Name() const
    {
        return name_;
    }

private:
    std::string name_;                 // Имя функции для GUI, таблиц и отчетов.
    std::string expression_;           // Оригинальный текст выражения.
    std::size_t dimension_;            // Размерность функции.

    mutable std::vector<double> variables_;   // Массив значений x1..xn, к которым привязан parser.
    mutable mu::Parser parser_;               // Экземпляр muparser для компиляции и вычисления выражения.
};

// ============================================================================
// МЕТОД MuParserObjectiveParser::Parse - СОЗДАНИЕ ГОТОВОЙ ЦЕЛЕВОЙ ФУНКЦИИ
// ============================================================================
// Принимает:
// - expression: текст математической формулы пользователя;
// - dimension: ожидаемая размерность пространства;
// - name: имя функции для GUI и таблиц;
// Записывает:
// - output: указатель на созданный объект функции при успехе;
// - error: текст ошибки при неудаче.
// Возвращает:
// - true, если функция успешно создана;
// - false, если выражение, размерность или инициализация некорректны.
// ============================================================================
bool MuParserObjectiveParser::Parse(const std::string& expression,
                                    std::size_t dimension,
                                    const std::string& name,
                                    IObjectiveFunction*& output,
                                    std::string& error) const
{
    // Сразу обнуляем выходной указатель.
    // Это важно, чтобы вызывающий код не получил "висячий" адрес при раннем возврате.
    output = NULL;

    // Очищаем предыдущий текст ошибки.
    error.clear();

    // Нулевая размерность для многомерной задачи запрещена.
    if (dimension == 0)
    {
        error = "Размерность должна быть положительной";
        return false;
    }

    // Дополнительно ограничиваем parser размерностью проекта.
    if (dimension > MAX_PARSER_DIMENSION)
    {
        error = "Парсер поддерживает размерность не больше 3";
        return false;
    }

    // Создаем конкретный объект функции динамически.
    // new здесь используется осознанно, потому что контракт Parse(...) возвращает
    // наружу именно "сырой" указатель IObjectiveFunction*.
    ParsedMuParserFunction* function = new ParsedMuParserFunction();

    // Если внутренняя инициализация не удалась, освобождаем объект вручную и
    // возвращаем false, чтобы не получить утечку памяти.
    if (!function->Initialize(name, expression, dimension, error))
    {
        delete function;
        return false;
    }

    // Передаем владение вызывающей стороне через выходной указатель.
    output = function;
    return true;
}

// ----------------------------------------------------------------------------
// ФУНКЦИЯ ANNOTATELASTPASSROWS
// ----------------------------------------------------------------------------
// Назначение:
// Отметить последние n строк трассировки, которые соответствуют только что
// завершившемуся полному проходу по всем направлениям.
//
// Зачем это нужно:
// Сам математический алгоритм оперирует проходами по базису направлений.
// Однако в таблице GUI каждая строка соответствует одному отдельному шагу
// по одному направлению s_j. Поэтому, когда в конце полного прохода
// фиксируется:
// - откат,
// - перестройка базиса,
// - новый базис,
// эту информацию нужно аккуратно проставить в уже добавленные строки.
//
// Принимает:
// - result            : итоговую структуру с накопленной трассировкой;
// - dimension         : размерность задачи, равная числу направлений в проходе;
// - rollback          : надо ли отметить факт отката;
// - directionChanged  : надо ли отметить факт перестройки направлений;
// - newBasis          : текстовое представление нового базиса;
// - newBasisDirections: сами новые направления для графика;
// - newBasisOrigin    : точку, из которой базис нужно рисовать.
//
// Изменяет:
// - последние dimension строк массива result.trace.
//
// Возвращает:
// - ничего.
// ----------------------------------------------------------------------------
static void AnnotateLastPassRows(OptimizationResult& result,
                                 std::size_t dimension,
                                 bool rollback,
                                 bool directionChanged,
                                 const std::string& newBasis,
                                 const std::vector<Vector>& newBasisDirections,
                                 const Vector& newBasisOrigin)
{
    // Если в трассировке пока меньше строк, чем размерность задачи, то
    // завершенного полного прохода еще просто нет. В такой ситуации ничего
    // размечать не нужно.
    if (result.trace.size() < dimension)
    {
        return;
    }

    // Последний полный проход состоит ровно из dimension строк.
    // Поэтому его первая строка находится по индексу:
    //     trace.size() - dimension.
    const std::size_t start = result.trace.size() - dimension;

    for (std::size_t i = 0; i < dimension; ++i)
    {
        // Берем ссылку именно на существующую строку в trace, чтобы менять
        // данные прямо внутри массива, а не работать с временной копией.
        OptimizationResult::TraceRow& row = result.trace[start + i];

        // Визуально откат и перестройка базиса фиксируются не у каждой строки
        // прохода, а на последней строке этого прохода. Так легче читать
        // таблицу: видно, что событие произошло именно по завершении полного
        // обхода всех направлений.
        const bool isLastRowInPass = (i + 1 == dimension);

        // rollback и directionChanged приходят сюда как признаки, относящиеся
        // ко всему завершенному проходу по базису.
        //
        // Но в таблице мы намеренно показываем эти события только на
        // последней строке прохода, чтобы:
        // 1) не дублировать один и тот же факт в каждой строке;
        // 2) визуально привязать событие к моменту завершения полного обхода
        //    всех направлений;
        // 3) сделать таблицу ближе к логике методички, где откат или
        //    перестройка базиса фиксируются после проверки всего прохода.
        row.rollback = (rollback && isLastRowInPass);
        row.directionChanged = (directionChanged && isLastRowInPass);

        if (row.directionChanged)
        {
            // Если на последней строке прохода произошла перестройка базиса,
            // сохраняем:
            // 1) строковое представление базиса;
            // 2) сами новые направления;
            // 3) точку начала нового базиса.
            row.newBasis = newBasis;
            row.newBasisDirections = newBasisDirections;
            row.newBasisOrigin = newBasisOrigin;
        }
        else
        {
            // Иначе очищаем эти поля, чтобы в таблице не "тянулись" старые
            // значения от предыдущих проходов.
            row.newBasis.clear();
            row.newBasisDirections.clear();
            row.newBasisOrigin.clear();
        }
    }
}

// ----------------------------------------------------------------------------
// ФУНКЦИЯ FORMATVECTORSHORT
// ----------------------------------------------------------------------------
// Назначение:
// Преобразовать числовой вектор в компактную строку вида "(a, b, c)".
//
// Зачем это нужно:
// GUI-таблица и итоговые строки требуют короткого текстового представления
// точек, направлений и наборов шагов.
//
// Принимает:
// - values : вектор чисел для форматирования.
//
// Возвращает:
// - строку с числами, записанными с 4 знаками после запятой.
// ----------------------------------------------------------------------------
static std::string FormatVectorShort(const Vector& values)
{
    // Поток создается локально внутри функции, потому что строка собирается
    // заново при каждом вызове и не должна хранить состояние между вызовами.
    std::ostringstream out;

    // Сначала открываем круглую скобку.
    out << "(";

    for (std::size_t i = 0; i < values.size(); ++i)
    {
        // Начиная со второго элемента, добавляем разделитель.
        if (i > 0)
        {
            out << ", ";
        }

        // std::fixed фиксирует формат вещественных чисел, а setprecision(4)
        // задает 4 знака после запятой.
        out << std::fixed << std::setprecision(4) << values[i];
    }

    // Закрываем круглую скобку и возвращаем собранную строку.
    out << ")";
    return out.str();
}

// ----------------------------------------------------------------------------
// ФУНКЦИЯ FORMATBASIS
// ----------------------------------------------------------------------------
// Назначение:
// Сформировать строку с описанием базиса направлений в виде
// "[s1=(...), s2=(...), ...]".
//
// Принимает:
// - directions : набор направлений текущего или нового базиса.
//
// Возвращает:
// - строковое представление всего базиса.
// ----------------------------------------------------------------------------
static std::string FormatBasis(const std::vector<Vector>& directions)
{
    // ostringstream — это поток для записи в строку.
    // Он удобен тем, что позволяет собирать итоговую строку по частям почти
    // так же, как если бы мы печатали данные в cout.
    std::ostringstream out;

    // Открываем квадратную скобку, чтобы получить формат:
    //     [s1=(...), s2=(...), ...]
    out << "[";

    for (std::size_t i = 0; i < directions.size(); ++i)
    {
        // Разделитель "; " нужен между направлениями, но не перед первым
        // элементом, поэтому проверка стоит именно на i > 0.
        if (i > 0)
        {
            out << "; ";
        }

        // Здесь последовательно записываем:
        // 1) имя направления s1, s2, ...;
        // 2) знак "=";
        // 3) компактное строковое представление самого вектора.
        out << "s" << (i + 1) << "=" << FormatVectorShort(directions[i]);
    }

    // Закрываем квадратную скобку и возвращаем готовую строку.
    out << "]";
    return out.str();
}

// ============================================================================
// КОНСТРУКТОРЫ СТРУКТУР ДАННЫХ
// ============================================================================
// Ниже размещены простые конструкторы, которые создают корректные начальные
// значения для опций метода, запроса, строки трассировки и итогового ответа.
// ============================================================================

RosenbrockDiscreteOptions::RosenbrockDiscreteOptions()
    : epsilon(DEFAULT_EPSILON),
      alpha(DEFAULT_ALPHA),
      beta(DEFAULT_BETA),
      maxIterations(DEFAULT_MAX_ITERATIONS)
{
    // Список инициализации выполняется раньше, чем тело конструктора.
    // В C++ это правильный способ задавать начальные значения полям класса.
    //
    // Здесь мы сразу получаем полностью готовый набор стандартных параметров
    // метода:
    // - epsilon       : точность по умолчанию;
    // - alpha         : коэффициент растяжения;
    // - beta          : коэффициент сжатия;
    // - maxIterations : защитный лимит числа внешних итераций.
}

OptimizationRequest::OptimizationRequest()
    : function(NULL)
{
    // По умолчанию указатель на функцию равен NULL.
    // Это явно показывает, что без отдельной инициализации запрос еще не
    // содержит целевую функцию и не готов к запуску Optimize(...).
}

OptimizationResult::TraceRow::TraceRow()
    : k(0),
      j(0),
      fxk(0.0),
      fyi(0.0),
      delta(0.0),
      fTrial(0.0),
      successfulStep(false),
      rollback(false),
      directionChanged(false)
{
    // Все поля строки трассировки инициализируются нейтральными значениями:
    // - индексы k и j равны нулю;
    // - числовые величины равны 0.0;
    // - логические признаки шага выключены.
    //
    // Это важно, потому что строка сначала создается пустой, а потом поля
    // постепенно заполняются в Optimize(...). Нейтральная инициализация
    // защищает от случайного чтения неинициализированных значений.
}

OptimizationResult::OptimizationResult()
    : converged(false),
      optimumValue(0.0),
      iterations(0),
      functionEvaluations(0)
{
    // Итоговая структура результата тоже стартует из нейтрального состояния:
    // - сходимость еще не достигнута;
    // - оптимальное значение пока неизвестно;
    // - число итераций и вычислений функции равно нулю.
    //
    // Остальные составные поля, такие как векторы и строки, будут
    // автоматически инициализированы своими пустыми значениями.
}

// ============================================================================
// МЕТОД OPTIMIZE
// ============================================================================
// Назначение:
// Реализовать дискретный метод Розенброка в точном соответствии с той схемой,
// которую пользователь привел из методички.
//
// ВАЖНО:
// Шаги алгоритма из методички ниже не выписаны отдельным большим блоком,
// а вставлены прямо по месту исполнения кода. Это сделано специально:
// 1) чтобы студент мог открыть методичку и идти по коду сверху вниз;
// 2) чтобы было видно, какая именно строчка реализует конкретный шаг схемы;
// 3) чтобы теория и реализация читались как единый текст, а не как две
//    слабо связанные части.
//
// Принимает:
// - request : полный запрос на оптимизацию.
//
// Записывает:
// - result : подробный результат метода и вся трассировка;
// - error  : текст ошибки, если выполнить метод не удалось.
//
// Возвращает:
// - true, если метод удалось корректно запустить и завершить;
// - false, если возникла ошибка входных данных или вычисления функции.
// ============================================================================
bool RosenbrockDiscreteOptimizer::Optimize(const OptimizationRequest& request,
                                           OptimizationResult& result,
                                           std::string& error) const
{
    // Сразу очищаем выходные структуры, чтобы не осталось мусора от прошлого
    // запуска, если тот же объект используется повторно.
    //
    // Это обычная техническая инициализация реализации. К математической части
    // метода Розенброка она не относится, но без нее в result могли бы
    // сохраниться старые строки трассировки, старые значения optimumX и т.д.
    result = OptimizationResult();
    error.clear();

    // Метод не может работать без целевой функции.
    //
    // В языке C++ указатель может быть равен NULL. Это означает, что объект,
    // на который он должен был указывать, отсутствует. Разыменовывать такой
    // указатель нельзя: это привело бы к неопределенному поведению.
    if (request.function == NULL)
    {
        error = "Функция оптимизации не задана";
        return false;
    }

    // Проверяем численные параметры метода: epsilon, alpha, beta,
    // maxIterations.
    if (!ValidateOptions(request.options, error))
    {
        // Если параметры метода противоречат методичке, дальнейшие вычисления
        // не имеют смысла и метод немедленно завершает работу с ошибкой.
        return false;
    }

    // Размерность задачи берется у самой функции.
    const std::size_t dimension = request.function->Dimension();

    if (dimension == 0)
    {
        // Нулевая размерность не имеет смысла для задачи оптимизации: в таком
        // "пространстве" нет ни направлений e_j, ни координат начальной точки.
        error = "Размерность функции должна быть положительной";
        return false;
    }

    // Начальная точка обязана иметь ту же размерность, что и функция.
    if (request.initialPoint.size() != dimension)
    {
        error = "Размерность начальной точки не совпадает с размерностью функции";
        return false;
    }

    // ШАГ 0.
    // ------------------------------------------------------------------------
    // Методичка требует задать:
    // - ε > 0;
    // - α > 1;
    // - β ∈ (-1, 0);
    // - s_j = e_j;
    // - \bar(Δ_j) >= 0;
    // - x_1.
    //
    // К этому моменту:
    // - параметры ε, α, β уже проверены функцией ValidateOptions(...);
    // - x_1 уже лежит в request.initialPoint;
    // - теперь нужно построить \bar(Δ_j), то есть исходные длины шагов.
    //
    // Если пользователь не задал покомпонентные шаги, ниже будет использован
    // стандартный набор одинаковых шагов.
    Vector initialStepMagnitudes;
    if (!BuildInitialStepMagnitudes(request.options,
                                    dimension,
                                    initialStepMagnitudes,
                                    error))
    {
        return false;
    }

    // По методичке в начале:
    //     Δ_j = \bar(Δ_j).
    // Поэтому рабочие шаги deltas сразу инициализируем начальными шагами.
    //
    // Здесь важно понимать математический смысл:
    // - initialStepMagnitudes — это "эталонные" стартовые шаги \bar(Δ_j);
    // - deltas                — это текущие рабочие шаги Δ_j, которые будут
    //                           растягиваться через α и сжиматься через β.
    Vector deltas = initialStepMagnitudes;

    // По методичке:
    //     s_j = e_j, j = 1, ..., n.
    // То есть начальный базис — стандартный ортонормированный базис.
    //
    // В частности:
    // - в 2D получаем (1, 0) и (0, 1);
    // - в 3D получаем (1, 0, 0), (0, 1, 0), (0, 0, 1).
    std::vector<Vector> directions = IdentityDirections(dimension);

    // Счетчик вычислений функции нужен для итоговой статистики.
    //
    // Для численных методов это важный показатель, потому что стоимость
    // одного вычисления функции в реальных задачах может быть высокой.
    std::size_t functionEvaluations = 0;

    // По методичке:
    //     x_1 — начальная точка.
    Vector xk = request.initialPoint;

    // Значение функции в базовой точке x_k.
    double fxk = 0.0;
    if (!EvaluateObjective(request.function, xk, fxk, functionEvaluations, error))
    {
        // Без значения функции в стартовой точке нельзя запустить ни один
        // дальнейший шаг метода, потому что все сравнения строятся именно
        // относительно f(x_k) и f(y_j).
        return false;
    }

    // По методичке:
    //     y_1 = x_1.
    Vector y1 = xk;
    double fy1 = fxk;

    // Внешний счетчик итераций метода.
    // Он соответствует индексу k в методичке.
    //
    // Важно не путать:
    // - k — номер внешней итерации метода;
    // - j — номер шага внутри текущего прохода по направлениям;
    // - trace.size() — число строк в таблице, то есть число всех отдельных
    //                  пробных шагов по направлениям.
    std::size_t k = 1;

    // Внешний цикл соответствует последовательным внешним итерациям метода.
    // Формально шаг 1 внутри алгоритма может возвращаться назад много раз,
    // поэтому здесь удобнее ограничивать число внешних проходов защитным
    // лимитом maxIterations.
    for (std::size_t directionPass = 1;
         directionPass <= request.options.maxIterations;
         ++directionPass)
    {
        // directionPass нужен только как технический предохранитель от
        // бесконечного цикла. Сам математический индекс метода — это k.
        (void)directionPass;

        // В начале каждого полного прохода по направлениям считаем, что
        // текущая базисная точка внутреннего цикла равна y_1.
        //
        // То есть внутренний цикл всегда стартует не с x_k, а с текущей точки
        // y_1, которая внутри одной внешней итерации может уже несколько раз
        // улучшаться после удачных проходов по всем направлениям.
        Vector yj = y1;
        double fyj = fy1;

        // Для каждой строки трассировки удобно хранить "снимок" текущей
        // внешней базовой точки x_k и ее значения. Они не должны меняться
        // внутри данного полного прохода.
        const Vector xkSnapshot = xk;
        const double fxkSnapshot = fxk;

        // ШАГ 1 + ШАГ 2.
        // --------------------------------------------------------------------
        // Шаг 1 методички описывает проверку одного направления s_j.
        // Шаг 2 говорит: если j < n, то надо перейти к следующему направлению.
        //
        // В языке C++ это естественно реализуется циклом for по j.
        for (std::size_t j = 0; j < dimension; ++j)
        {
            // Создаем новую строку трассировки для одного конкретного шага по
            // одному конкретному направлению s_j.
            OptimizationResult::TraceRow row;

            // В таблице используем 1-based нумерацию, как в методичке.
            row.k = k;
            row.j = j + 1;

            // Фиксируем внешний базис x_k и его значение.
            row.xk = xkSnapshot;
            row.fxk = fxkSnapshot;

            // Фиксируем внутреннюю точку y_j и ее значение f(y_j).
            row.yi = yj;
            row.fyi = fyj;

            // Текущая длина шага Δ_j и текущее направление s_j.
            row.delta = deltas[j];
            row.direction = directions[j];

            // Для графики и таблицы также сохраняем весь текущий базис и точку
            // его привязки.
            //
            // Это уже не математическая необходимость метода, а сервисная
            // информация для визуализации: 2D и 3D viewer должны понимать,
            // каким базисом метод пользовался на этом конкретном шаге.
            row.basisDirections = directions;
            row.basisOrigin = yj;

            // Строим пробную точку:
            //     y_j + Δ_j s_j.
            //
            // Почему это делается через target + scale * direction, а не одной
            // формулой в отдельном выражении:
            // 1) так переиспользуется helper AddScaled(...);
            // 2) так проще читать численную операцию пошагово;
            // 3) так меньше промежуточных временных объектов.
            Vector candidate = yj;
            AddScaled(candidate, deltas[j], directions[j]);

            // Считаем значение функции в пробной точке.
            //
            // После этого у нас есть все данные, чтобы сравнить:
            // - f(y_j + Δ_j s_j)
            // - f(y_j)
            double fCandidate = 0.0;
            if (!EvaluateObjective(request.function,
                                   candidate,
                                   fCandidate,
                                   functionEvaluations,
                                   error))
            {
                // Ошибка вычисления функции в пробной точке делает текущий шаг
                // и всю итерацию недействительными, поэтому выходим сразу.
                return false;
            }

            // Сохраняем пробную точку y_j + Δ_j s_j в строку трассировки.
            // Она нужна:
            // 1) для табличного отображения;
            // 2) для графического показа пробного шага;
            // 3) для последующего анализа, какой именно кандидат сравнивался
            //    с текущей точкой y_j.
            row.trialPoint = candidate;

            // Сохраняем значение функции в пробной точке.
            // Это именно f(y_j + Δ_j s_j), то есть та величина, которая
            // непосредственно участвует в проверке успешности или неудачности
            // текущего шага метода.
            row.fTrial = fCandidate;

            // ШАГ 1.
            // ----------------------------------------------------------------
            // Если f(y_j + Δ_j s_j) < f(y_j), шаг успешный:
            //     y_{j+1} = y_j + Δ_j s_j,
            //     Δ_j = αΔ_j.
            if (fCandidate < fyj)
            {
                // Успех означает, что новая базисная точка внутреннего цикла
                // действительно сместилась вперед по направлению s_j.
                yj = candidate;

                // Одновременно обновляем значение функции в новой точке.
                fyj = fCandidate;

                // Растягиваем шаг именно по текущему направлению j.
                // Остальные deltas[i] при i != j не меняются.
                deltas[j] *= request.options.alpha;
                row.successfulStep = true;
            }
            else
            {
                // Иначе шаг неудачный:
                //     y_{j+1} = y_j,
                //     Δ_j = βΔ_j.
                //
                // Здесь yj и fyj не меняются специально. Это и есть запись
                // y_{j+1} = y_j в программной форме.
                deltas[j] *= request.options.beta;
                row.successfulStep = false;
            }

            // Добавляем строку в трассировку уже после того, как все поля
            // заполнены.
            //
            // Это важно для GUI: строка должна отражать законченное решение по
            // текущему направлению, а не промежуточное состояние до сравнения.
            result.trace.push_back(row);
        }

        // После завершения цикла по j переменные:
        // - yj  содержат y_{n+1};
        // - fyj содержат f(y_{n+1}).

        // ШАГ 3.
        // --------------------------------------------------------------------
        // Если после полного прохода удалось получить снижение относительно y_1,
        // то:
        //     y_1 = y_{n+1}
        // и алгоритм снова идет на шаг 1 с тем же набором направлений.
        //
        // Иначе говоря:
        // - внешний базис x_k пока не меняется;
        // - направления s_j пока не перестраиваются;
        // - просто берется новая более удачная точка y_1 и начинается еще один
        //   полный проход по тем же направлениям.
        if (fyj < fy1)
        {
            // По шагу 3 методички новая внутренняя базисная точка y_1
            // приравнивается к y_{n+1}, найденной после полного прохода по
            // всем направлениям.
            y1 = yj;

            // Одновременно обновляем и значение функции в этой новой точке,
            // чтобы на следующем проходе не вычислять его повторно.
            fy1 = fyj;

            // Для GUI обновляем номер внешней итерации, хотя k пока не
            // менялся. Это именно текущая внешняя итерация.
            result.iterations = k;

            // В этом случае не было ни отката, ни перестройки базиса.
            AnnotateLastPassRows(result,
                                 dimension,
                                 false,
                                 false,
                                 "",
                                 std::vector<Vector>(),
                                 Vector());

            // continue здесь означает буквальный переход к новому полному
            // проходу по направлениям с тем же самым базисом.
            //
            // То есть:
            // - мы не перестраиваем направления;
            // - не фиксируем новый внешний базис x_{k+1};
            // - не выполняем критерий остановки из шага 5;
            // - а сразу возвращаемся к началу внешнего цикла, где новая y_1
            //   станет стартовой точкой очередного прохода.
            continue;
        }

        // ШАГ 4 и ШАГ 5.
        // --------------------------------------------------------------------
        // Если полного улучшения относительно y_1 уже нет, но при этом
        // f(y_{n+1}) < f(x_k), значит накопленное улучшение относительно
        // внешней базовой точки x_k все же есть. Тогда переходим к шагу 5.
        //
        // Это очень важный момент метода:
        // - последний полный проход сам по себе уже не дал улучшения
        //   относительно текущего y_1;
        // - но вся внешняя итерация от x_k до y_{n+1} оказалась успешной;
        // - значит настало время зафиксировать новый внешний базис x_{k+1}
        //   и перестроить направления.
        if (fyj < fxk)
        {
            // По шагу 5:
            //     x_{k+1} = y_{n+1}.
            Vector xk1 = yj;
            double fxk1 = fyj;

            // Вычисляем смещение между новой и старой внешней базовой точкой.
            Vector displacement = Subtract(xk1, xk);

            // ШАГ 5, критерий остановки:
            //     если ||x_{k+1} - x_k|| < ε,
            //     то метод завершен.
            if (Norm(displacement) < request.options.epsilon)
            {
                // Фиксируем факт успешного завершения метода.
                result.converged = true;

                // Текст сообщения нужен для GUI, итогового отчета и PNG-
                // выгрузки. Он поясняет, какой именно критерий остановки
                // сработал в данном запуске.
                result.message = "Достигнута сходимость: ||x(k+1)-x(k)|| < эпсилон";

                // По шагу 5 ответом становится именно новая внешняя базовая
                // точка x_{k+1}.
                result.optimumX = xk1;
                result.optimumValue = fxk1;

                // Сохраняем текущий набор шагов Δ_j на момент остановки.
                // Это полезно для таблицы, итогового блока GUI и анализа
                // того, насколько "мелкими" стали шаги по направлениям.
                result.finalStepMagnitudes = deltas;

                // iterations здесь хранит текущий номер внешней итерации k,
                // на которой метод завершился.
                result.iterations = k;

                // Сохраняем итоговое число реальных вычислений функции.
                result.functionEvaluations = functionEvaluations;

                AnnotateLastPassRows(result,
                                     dimension,
                                     false,
                                     false,
                                     "",
                                     std::vector<Vector>(),
                                     Vector());

                // Возвращаем true, потому что метод завершился штатно:
                // не по ошибке и не по аварийному выходу, а по теоретическому
                // критерию остановки.
                return true;
            }

            // Если остановки нет, нужно вычислить коэффициенты λ_j из
            // разложения:
            //     x_{k+1} - x_k = Σ λ_j s_j.
            //
            // Так как directions образуют ортонормированный базис,
            // коэффициенты λ_j находятся обычными скалярными произведениями.
            //
            // Если бы базис не был ортонормированным, пришлось бы решать
            // отдельную линейную систему. Здесь же ортонормированность сильно
            // упрощает вычисления.
            Vector lambdas(dimension, 0.0);
            for (std::size_t j = 0; j < dimension; ++j)
            {
                // Поскольку текущий базис directions ортонормирован,
                // коэффициент λ_j равен проекции вектора смещения
                // (x_{k+1} - x_k) на направление s_j.
                //
                // В ортонормированном базисе такая проекция вычисляется простым
                // скалярным произведением, поэтому отдельную систему уравнений
                // решать не нужно.
                lambdas[j] = Dot(displacement, directions[j]);
            }

            // По формулам (3.1) строим новый набор направлений.
            std::vector<Vector> newDirections;
            if (!BuildNewDirections(directions, lambdas, newDirections, error))
            {
                // Если новый базис построить не удалось, то продолжать метод
                // уже нельзя: без корректного набора направлений следующие
                // шаги потеряют математический смысл.
                return false;
            }

            // Сразу формируем строковое представление нового базиса для таблицы.
            const std::string newBasis = FormatBasis(newDirections);

            // Обновляем рабочий базис направлений.
            directions = newDirections;

            // ШАГ 6.
            // ----------------------------------------------------------------
            // После перестройки базиса нужно:
            // 1) сбросить Δ_j к исходным \bar(Δ_j);
            // 2) положить y_1 = x_{k+1};
            // 3) увеличить k.
            //
            // Смысл сброса deltas:
            // старые длины шагов накапливались под старый базис направлений.
            // После перестройки базиса они уже не имеют прежнего смысла, и
            // методичка требует начать работу по новому базису заново с
            // исходных \bar(Δ_j).
            deltas = initialStepMagnitudes;

            // Теперь внешний базис метода официально переносится в x_{k+1}.
            // Начиная со следующего полного прохода именно эта точка будет
            // играть роль x_k.
            xk = xk1;
            fxk = fxk1;

            // Внутренняя базисная точка y_1 после шага 6 тоже должна быть
            // равна x_{k+1}. Это уже старт следующей внешней итерации.
            y1 = xk1;
            fy1 = fxk1;

            // Увеличиваем номер внешней итерации метода:
            // старое k соответствовало предыдущему x_k, а теперь мы перешли
            // к новому базису x_{k+1}.
            ++k;

            // В result.iterations всегда хранится актуальный номер внешней
            // итерации, чтобы GUI мог сразу отображать его в таблице и
            // итоговом блоке.
            result.iterations = k;

            // Для таблицы и графика отмечаем, что по завершении последнего
            // полного прохода произошла перестройка базиса.
            AnnotateLastPassRows(result,
                                 dimension,
                                 false,
                                 true,
                                 newBasis,
                                 newDirections,
                                 xk1);
            continue;
        }

        // ШАГ 4, альтернативная ветвь.
        // --------------------------------------------------------------------
        // Если дошли сюда, то полного улучшения относительно x_k нет.
        // По методичке надо проверить:
        //     |Δ_j| < ε для всех j ?
        //
        // Если ответ "да", метод исчерпал возможности локального улучшения при
        // текущей точности и завершается в x_k.
        if (AllStepMagnitudesBelowEpsilon(deltas, request.options.epsilon))
        {
            // Фиксируем, что метод завершился по второму критерию остановки:
            // все рабочие шаги стали меньше заданной точности epsilon.
            result.converged = true;

            // Сообщение явно указывает, что сработал критерий по |Δ_j|.
            result.message = "Достигнута сходимость: |delta_j| < эпсилон для всех направлений";

            // По шагу 4 в этой ветви ответом остается текущая внешняя
            // базовая точка x_k, потому что нового x_{k+1} не зафиксировано.
            result.optimumX = xk;
            result.optimumValue = fxk;

            // Сохраняем итоговый набор шагов по всем направлениям.
            result.finalStepMagnitudes = deltas;

            // Запоминаем номер внешней итерации и число вычислений функции на
            // момент остановки.
            result.iterations = k;
            result.functionEvaluations = functionEvaluations;

            AnnotateLastPassRows(result,
                                 dimension,
                                 false,
                                 false,
                                 "",
                                 std::vector<Vector>(),
                                 Vector());

            // Возвращаем true, потому что это нормальное завершение метода по
            // теоретическому критерию, а не авария.
            return true;
        }

        // Если шаги еще недостаточно малы, по методичке нужно:
        //     y_1 = y_{n+1}, j = 1 и переход на шаг 1.
        //
        // В нашем коде это реализуется простым обновлением y1/fy1 и
        // переходом к следующей итерации внешнего цикла.
        //
        // То есть:
        // - направления пока остаются прежними;
        // - внешний базис x_k пока не фиксируется заново;
        // - метод просто продолжает поиск с новой внутренней точки y_1.
        y1 = yj;
        fy1 = fyj;

        result.iterations = k;

        // Здесь по завершении прохода фиксируем откат. В терминах GUI это
        // означает, что по последнему полному проходу не было успешного
        // обновления внешней базовой точки x_k.
        AnnotateLastPassRows(result,
                             dimension,
                             true,
                             false,
                             "",
                             std::vector<Vector>(),
                             Vector());
    }

    // Если внешний защитный лимит итераций исчерпан, возвращаем частичный
    // результат с пометкой о несходимости. Это технический предохранитель, а
    // не шаг методички.
    //
    // Такой блок нужен любой практической программе: если пользователь
    // подаст плохие параметры или сложную функцию, приложение не должно
    // зависнуть бесконечно.
    result.converged = false;
    result.message = "Достигнуто максимальное число итераций";
    result.functionEvaluations = functionEvaluations;
    result.iterations = k;
    result.finalStepMagnitudes = deltas;

    // В качестве лучшей найденной точки возвращаем либо y_1, либо x_k —
    // смотря где значение функции меньше.
    if (fy1 < fxk)
    {
        // Если внутренняя точка y_1 на момент выхода по лимиту лучше,
        // чем текущий внешний базис x_k, то разумно вернуть именно ее.
        // Иначе пользователь потерял бы более удачное найденное значение.
        result.optimumX = y1;
        result.optimumValue = fy1;
    }
    else
    {
        // В противном случае лучшей остается точка x_k.
        result.optimumX = xk;
        result.optimumValue = fxk;
    }

    // Несмотря на то, что сходимость не достигнута, функция возвращает true:
    // сама процедура метода отработала корректно, просто остановилась по
    // техническому лимиту maxIterations.
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД VALIDATEOPTIONS
// ----------------------------------------------------------------------------
// Назначение:
// Проверить корректность численных параметров метода.
//
// Принимает:
// - options : набор параметров дискретного метода Розенброка.
//
// Записывает:
// - error : текст причины ошибки, если параметры некорректны.
//
// Возвращает:
// - true, если параметры допустимы;
// - false, если найдено нарушение ограничений методички.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::ValidateOptions(const RosenbrockDiscreteOptions& options,
                                                  std::string& error)
{
    // error намеренно не очищается автоматически внутри каждого if-блока,
    // потому что при первой найденной проблеме функция сразу завершится.
    // Следовательно, в error попадет только одна, но самая первая и самая
    // важная причина некорректности входных параметров.

    // По методичке epsilon должен быть строго положительным.
    if (!(options.epsilon > 0.0))
    {
        // Сразу записываем в error понятное человеку сообщение, чтобы GUI мог
        // показать не просто факт ошибки, а ее причину.
        error = "Эпсилон должен быть положительным";
        return false;
    }

    // По методичке alpha > 1.
    if (!(options.alpha > 1.0))
    {
        // При alpha <= 1 успешный шаг не растягивался бы, а либо оставался
        // прежним, либо даже уменьшался. Это уже не соответствует алгоритму.
        error = "Альфа должна быть больше 1";
        return false;
    }

    // По методичке beta принадлежит интервалу (-1, 0).
    if (!(options.beta > -1.0 && options.beta < 0.0))
    {
        // Здесь одновременно проверяются обе границы открытого интервала:
        // beta не может быть ни равен -1/0, ни выходить за пределы между ними.
        error = "Бета должна принадлежать интервалу (-1, 0)";
        return false;
    }

    // Это не пункт методички, а технический защитный параметр реализации.
    if (options.maxIterations == 0)
    {
        error = "Максимальное число итераций должно быть положительным";
        return false;
    }

    // Если ни одна проверка не сработала, параметры считаются корректными и
    // метод может безопасно стартовать.
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД BUILDINITIALSTEPMAGNITUDES
// ----------------------------------------------------------------------------
// Назначение:
// Построить набор начальных длин шагов \bar(Δ_j).
//
// Логика:
// - если пользователь не передал покомпонентный набор шагов, то используем
//   одинаковые шаги величины 1.0 по всем направлениям;
// - если набор передан, проверяем его размер и корректность значений.
//
// Принимает:
// - options   : параметры метода;
// - dimension : размерность задачи.
//
// Записывает:
// - steps : итоговый набор начальных длин шагов;
// - error : текст ошибки.
//
// Возвращает:
// - true при успехе;
// - false при некорректных данных.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::BuildInitialStepMagnitudes(
    const RosenbrockDiscreteOptions& options,
    std::size_t dimension,
    Vector& steps,
    std::string& error)
{
    // Сначала полностью очищаем steps, чтобы в выходном векторе не осталось
    // значений от предыдущего вызова этой же функции.
    steps.clear();

    // Если пользователь не задал индивидуальные шаги, используем
    // одинаковые шаги 1.0 по всем координатам.
    if (options.initialStepMagnitudes.empty())
    {
        // assign(dimension, 1.0) создает вектор нужной длины и заполняет все
        // его элементы одним и тем же значением 1.0.
        steps.assign(dimension, 1.0);

        // На этом работа функции заканчивается, потому что выходной массив
        // шагов уже полностью сформирован.
        return true;
    }

    // Количество шагов должно совпадать с размерностью задачи.
    if (options.initialStepMagnitudes.size() != dimension)
    {
        // Здесь проверяется не величина шага, а именно число шагов:
        // по одному начальному Delta_j на каждое направление пространства.
        error = "Количество начальных длин шагов должно совпадать с размерностью";
        return false;
    }

    for (std::size_t i = 0; i < options.initialStepMagnitudes.size(); ++i)
    {
        // Берем очередной пользовательский шаг в отдельную локальную
        // переменную, чтобы:
        // 1) не обращаться к вектору по индексу несколько раз;
        // 2) упростить чтение условия ниже;
        // 3) иметь осмысленное имя step в комментариях и отладке.
        const double step = options.initialStepMagnitudes[i];

        // Начальные длины шагов должны быть:
        // 1) конечными;
        // 2) строго положительными.
        if (!(step > 0.0) || !std::isfinite(step))
        {
            // std::isfinite(step) отбрасывает NaN и бесконечности, которые
            // численно опасны для метода.
            error = "Начальные длины шагов должны быть конечными и строго положительными";
            return false;
        }
    }

    // После всех проверок можно безопасно скопировать пользовательский набор
    // шагов в рабочий выходной вектор.
    //
    // Здесь выполняется именно копирование значений, а не привязка ссылкой:
    // дальнейшие изменения рабочего вектора steps не должны менять исходный
    // массив options.initialStepMagnitudes.
    steps = options.initialStepMagnitudes;

    // true сообщает вызывающему коду, что в steps лежит корректный набор
    // начальных длин шагов \u0304(Delta_j).
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД EVALUATEOBJECTIVE
// ----------------------------------------------------------------------------
// Назначение:
// Безопасно вычислить значение целевой функции и увеличить счетчик вычислений.
//
// Принимает:
// - function    : объект целевой функции;
// - x           : точка, в которой надо вычислить функцию.
//
// Записывает:
// - value       : найденное значение функции;
// - evalCounter : счетчик вычислений функции;
// - error       : текст ошибки при неудаче.
//
// Возвращает:
// - true, если вычисление прошло успешно;
// - false, если функция не задана или внутри Evaluate возникла ошибка.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::EvaluateObjective(const IObjectiveFunction* function,
                                                    const Vector& x,
                                                    double& value,
                                                    std::size_t& evalCounter,
                                                    std::string& error)
{
    if (function == NULL)
    {
        // Проверка дублируется и здесь, хотя основной вызов Optimize(...)
        // уже делает такую же валидацию. Это нормальная локальная защита
        // helper-функции от некорректного прямого вызова.
        error = "Функция оптимизации не задана";
        return false;
    }

    // Локальная строка нужна, чтобы не портить основной error, пока неясно,
    // завершится ли вызов function->Evaluate(...) успешно.
    std::string localError;

    // Реальная работа делегируется конкретной реализации IObjectiveFunction.
    if (!function->Evaluate(x, value, localError))
    {
        // Сообщение конкретной реализации функции сохраняем и поднимаем на
        // уровень выше, добавляя общий префикс для единообразия интерфейса.
        error = "Ошибка вычисления функции: " + localError;
        return false;
    }

    // Счетчик увеличиваем только при успешном вычислении.
    ++evalCounter;

    // true означает, что в value уже лежит корректное значение f(x), и
    // вызывающий код может безопасно использовать его в сравнениях.
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД DOT
// ----------------------------------------------------------------------------
// Назначение:
// Вычислить скалярное произведение двух векторов.
//
// Принимает:
// - a, b : векторы.
//
// Возвращает:
// - сумму произведений координат.
//
// Замечание:
// Если размеры векторов вдруг различаются, берется минимальная длина.
// Это защитное поведение реализации.
// ----------------------------------------------------------------------------
double RosenbrockDiscreteOptimizer::Dot(const Vector& a, const Vector& b)
{
    // Изначально предполагаем, что обработаем все координаты первого вектора.
    std::size_t n = a.size();
    if (b.size() < n)
    {
        // Если второй вектор короче, принудительно уменьшаем число
        // обрабатываемых координат до безопасного минимума.
        n = b.size();
    }

    double sum = 0.0;
    for (std::size_t i = 0; i < n; ++i)
    {
        // sum накапливает классическое скалярное произведение:
        // a1*b1 + a2*b2 + ... + an*bn.
        sum += a[i] * b[i];
    }

    // Возвращаем готовое скалярное произведение.
    return sum;
}

// ----------------------------------------------------------------------------
// МЕТОД NORM
// ----------------------------------------------------------------------------
// Назначение:
// Вычислить евклидову норму вектора.
//
// Принимает:
// - values : входной вектор.
//
// Возвращает:
// - sqrt(<values, values>).
// ----------------------------------------------------------------------------
double RosenbrockDiscreteOptimizer::Norm(const Vector& values)
{
    // Норма строится через уже реализованное скалярное произведение, чтобы не
    // дублировать одну и ту же формулу в нескольких местах файла.
    return std::sqrt(Dot(values, values));
}

// ----------------------------------------------------------------------------
// МЕТОД SUBTRACT
// ----------------------------------------------------------------------------
// Назначение:
// Вычислить разность двух векторов a - b.
//
// Принимает:
// - a : уменьшаемый вектор;
// - b : вычитаемый вектор.
//
// Возвращает:
// - новый вектор a - b.
//
// Замечание:
// Если b короче a, отсутствующие координаты b считаются равными нулю.
// ----------------------------------------------------------------------------
Vector RosenbrockDiscreteOptimizer::Subtract(const Vector& a, const Vector& b)
{
    // Сразу создаем вектор результата нужной длины и заполняем его нулями,
    // чтобы потом последовательно записывать координаты разности.
    Vector out(a.size(), 0.0);

    for (std::size_t i = 0; i < a.size(); ++i)
    {
        // Тернарный оператор "? :" в C++ — это компактная форма if/else для
        // выбора одного из двух выражений.
        //
        // Здесь он означает:
        // - если у b есть координата с индексом i, берем ее;
        // - иначе считаем недостающую координату равной нулю.
        const double right = (i < b.size()) ? b[i] : 0.0;

        // Записываем i-ю координату результата a - b.
        out[i] = a[i] - right;
    }

    // Возвращаем уже полностью собранный вектор разности.
    return out;
}

// ----------------------------------------------------------------------------
// МЕТОД ADDSCALED
// ----------------------------------------------------------------------------
// Назначение:
// Выполнить операцию:
//     target = target + scale * direction.
//
// Принимает:
// - scale     : коэффициент при direction;
// - direction : добавляемое направление.
//
// Изменяет:
// - target : целевой вектор.
//
// Замечание:
// Если размеры векторов отличаются, используется минимальная длина.
// ----------------------------------------------------------------------------
void RosenbrockDiscreteOptimizer::AddScaled(Vector& target,
                                            double scale,
                                            const Vector& direction)
{
    // По умолчанию хотим пройтись по всему target.
    std::size_t n = target.size();
    if (direction.size() < n)
    {
        // Как и в Dot(...), защищаемся от выхода за границы более короткого
        // вектора направления.
        n = direction.size();
    }

    for (std::size_t i = 0; i < n; ++i)
    {
        // Это прямая реализация формулы:
        //     target_i = target_i + scale * direction_i.
        target[i] += scale * direction[i];
    }
}

// ----------------------------------------------------------------------------
// МЕТОД ALLSTEPMAGNITUDESBELOWEPSILON
// ----------------------------------------------------------------------------
// Назначение:
// Проверить критерий остановки по всем длинам шагов:
//     |Δ_j| < ε для всех j.
//
// Принимает:
// - deltas  : набор текущих шагов;
// - epsilon : порог точности.
//
// Возвращает:
// - true, если все шаги достаточно малы;
// - false, если хотя бы один шаг еще не меньше epsilon.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::AllStepMagnitudesBelowEpsilon(const Vector& deltas,
                                                                double epsilon)
{
    for (std::size_t i = 0; i < deltas.size(); ++i)
    {
        if (std::abs(deltas[i]) >= epsilon)
        {
            // Достаточно найти хотя бы один шаг, который еще не стал меньше
            // epsilon, чтобы весь критерий считался невыполненным.
            return false;
        }
    }

    // Если цикл завершился без раннего return false, значит каждый шаг delta_i
    // по модулю оказался меньше epsilon.
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД IDENTITYDIRECTIONS
// ----------------------------------------------------------------------------
// Назначение:
// Построить начальный базис направлений:
//     s_j = e_j, j = 1, ..., n.
//
// Принимает:
// - dimension : размерность задачи.
//
// Возвращает:
// - массив стандартных базисных векторов.
// ----------------------------------------------------------------------------
std::vector<Vector> RosenbrockDiscreteOptimizer::IdentityDirections(std::size_t dimension)
{
    // reserve(...) заранее выделяет память под все направления и уменьшает
    // число возможных перераспределений памяти во время push_back(...).
    std::vector<Vector> directions;
    directions.reserve(dimension);

    for (std::size_t i = 0; i < dimension; ++i)
    {
        // Создаем очередной единичный вектор e_i.
        Vector basis(dimension, 0.0);

        // У единичного вектора e_i ровно одна координата равна 1.0, а все
        // остальные остаются нулями.
        basis[i] = 1.0;

        // push_back(...) дописывает готовый вектор в конец массива directions.
        directions.push_back(basis);
    }

    // На выходе directions содержит базис:
    // e_1, e_2, ..., e_n.
    return directions;
}

// ----------------------------------------------------------------------------
// МЕТОД ORTHOGONALIZEWITHFALLBACK
// ----------------------------------------------------------------------------
// Назначение:
// Ортогонализовать кандидат на новое направление и, если он выродился,
// аккуратно применить запасные варианты.
//
// Почему нужен fallback:
// В численных алгоритмах после вычитания проекций норма вектора может стать
// очень маленькой из-за:
// - близкой линейной зависимости;
// - накопления ошибок округления;
// - почти нулевых коэффициентов lambda.
//
// Тогда метод пробует:
// 1) исходный candidate;
// 2) старое направление oldDirections[index];
// 3) стандартный единичный вектор.
//
// Принимает:
// - candidate      : основной кандидат на новое направление;
// - basis          : уже построенная часть нового ортонормированного базиса;
// - oldDirections  : старый базис направлений;
// - index          : номер текущего направления;
//
// Записывает:
// - outDirection   : итоговое нормированное направление;
// - error          : текст ошибки при неудаче.
//
// Возвращает:
// - true, если удалось построить корректное направление;
// - false, если все варианты выродились.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::OrthogonalizeWithFallback(
    const Vector& candidate,
    const std::vector<Vector>& basis,
    const std::vector<Vector>& oldDirections,
    std::size_t index,
    Vector& outDirection,
    std::string& error)
{
    // Начинаем с основного кандидата.
    //
    // Копирование в отдельный вектор b нужно потому, что дальше мы будем
    // многократно вычитать проекции и менять координаты этого кандидата.
    Vector b = candidate;

    // Вычитаем проекции на уже построенные новые направления.
    //
    // Это и есть суть ортогонализации: мы удаляем из b все компоненты,
    // лежащие вдоль уже построенных базисных векторов, чтобы получить
    // направление, ортогональное предыдущим.
    for (std::size_t i = 0; i < index && i < basis.size(); ++i)
    {
        AddScaled(b, -Dot(b, basis[i]), basis[i]);
    }

    double norm = Norm(b);

    // norm — это текущая длина вектора b после ортогонализации.
    // Именно по ней мы понимаем, не выродился ли кандидат численно.

    // Если норма неадекватна, пробуем в качестве запасного варианта старое
    // направление с тем же индексом.
    if (!std::isfinite(norm) || norm <= DIRECTION_TOLERANCE)
    {
        // Здесь используется не копирование ссылки, а копирование самого
        // вектора. Это важно: ниже b будет модифицироваться, а старый базис
        // oldDirections должен остаться неизменным.
        b = oldDirections[index];

        for (std::size_t i = 0; i < index && i < basis.size(); ++i)
        {
            // Здесь повторяем ту же ортогонализацию, но уже для запасного
            // варианта направления.
            AddScaled(b, -Dot(b, basis[i]), basis[i]);
        }

        norm = Norm(b);
    }

    // Если и это не помогло, пробуем стандартный единичный вектор.
    if (!std::isfinite(norm) || norm <= DIRECTION_TOLERANCE)
    {
        // assign(...) полностью перезаписывает содержимое вектора b:
        // 1) меняет размер;
        // 2) заполняет все координаты нулями.
        b.assign(candidate.size(), 0.0);

        if (index < b.size())
        {
            // Нормальный случай: ставим единицу в координату с номером index.
            b[index] = 1.0;
        }
        else if (!b.empty())
        {
            // Защитная ветка на случай поврежденных размеров.
            b[0] = 1.0;
        }

        for (std::size_t i = 0; i < index && i < basis.size(); ++i)
        {
            AddScaled(b, -Dot(b, basis[i]), basis[i]);
        }

        norm = Norm(b);
    }

    if (!std::isfinite(norm) || norm <= DIRECTION_TOLERANCE)
    {
        // Если даже после всех fallback-вариантов норма осталась плохой,
        // значит корректный новый базис собрать не удалось.
        error = "Не удалось построить корректный ортонормированный набор направлений";
        return false;
    }

    // Нормируем вектор, чтобы получить единичное направление.
    for (std::size_t i = 0; i < b.size(); ++i)
    {
        // Делим каждую координату на одну и ту же норму, чтобы итоговая длина
        // вектора стала равна единице.
        b[i] /= norm;
    }

    // Переписываем подготовленный нормированный вектор в выходной параметр.
    outDirection = b;

    // true означает, что в outDirection записан корректный единичный вектор,
    // который можно использовать как направление нового базиса.
    return true;
}

// ----------------------------------------------------------------------------
// МЕТОД BUILDNEWDIRECTIONS
// ----------------------------------------------------------------------------
// Назначение:
// Построить новый базис направлений метода Розенброка по коэффициентам λ_j
// в соответствии с формулами (3.1).
//
// Идея:
// 1) сначала строятся промежуточные векторы a_j как нужные линейные
//    комбинации старых направлений;
// 2) затем из них строятся новые ортонормированные направления через
//    ортогонализацию типа Грама-Шмидта;
// 3) если численно возникает вырождение, применяется fallback.
//
// Принимает:
// - oldDirections : старый базис направлений;
// - lambdas       : коэффициенты разложения смещения по старому базису.
//
// Записывает:
// - newDirections : новый построенный базис;
// - error         : текст ошибки.
//
// Возвращает:
// - true при успешной перестройке;
// - false при невозможности построить новый базис.
// ----------------------------------------------------------------------------
bool RosenbrockDiscreteOptimizer::BuildNewDirections(
    const std::vector<Vector>& oldDirections,
    const Vector& lambdas,
    std::vector<Vector>& newDirections,
    std::string& error)
{
    // Сначала очищаем выходной массив, чтобы на случай раннего выхода по ошибке
    // в нем не осталось частично построенного старого базиса.
    newDirections.clear();

    if (oldDirections.empty())
    {
        // Перестраивать базис из пустого набора невозможно по определению.
        error = "Невозможно перестроить направления: старый набор пуст";
        return false;
    }

    const std::size_t n = oldDirections.size();
    const std::size_t vecSize = oldDirections[0].size();

    // n — число направлений в базисе.
    // vecSize — длина каждого вектора направления, то есть размерность
    // пространства задачи.

    if (lambdas.size() != n)
    {
        // Каждый коэффициент lambda_j должен соответствовать своему
        // направлению s_j, поэтому размеры обязаны совпадать.
        error = "Размер массива коэффициентов лямбда не совпадает с числом направлений";
        return false;
    }

    // Массив промежуточных векторов a_j из формул (3.1).
    std::vector<Vector> a(n, Vector(vecSize, 0.0));
    // Внешний std::vector хранит сразу все a_j, а внутренний Vector задает
    // нулевой шаблон одного промежуточного вектора нужной размерности.

    for (std::size_t j = 0; j < n; ++j)
    {
        // Если соответствующий lambda почти нулевой, нет смысла строить
        // длинную линейную комбинацию. В качестве безопасного варианта
        // сохраняем старое направление.
        if (std::abs(lambdas[j]) <= DIRECTION_TOLERANCE)
        {
            // При почти нулевом lambda_j линейная комбинация будет численно
            // ненадежной, поэтому безопаснее сохранить прежнее направление.
            a[j] = oldDirections[j];
            continue;
        }

        Vector sum(vecSize, 0.0);

        // sum будет накапливать линейную комбинацию старых направлений для a_j.

        // По схеме перестройки базиса a_j строится как сумма по хвосту
        // старых направлений с коэффициентами lambda.
        //
        // Иными словами, чем больше индекс i, тем "позднее" направление
        // старого базиса попадает в линейную комбинацию для a_j.
        for (std::size_t i = j; i < n; ++i)
        {
            // Накапливаем очередной член линейной комбинации вектора a_j.
            AddScaled(sum, lambdas[i], oldDirections[i]);
        }

        // После завершения внутреннего цикла sum уже содержит готовый a_j.
        a[j] = sum;
    }

    // Здесь будем накапливать уже ортонормированные новые направления.
    std::vector<Vector> updatedDirections(n, Vector(vecSize, 0.0));

    for (std::size_t j = 0; j < n; ++j)
    {
        // Начинаем ортогонализацию с текущего промежуточного вектора a_j.
        Vector b = a[j];
        // b специально отделен от a[j], потому что в процессе ортогонализации
        // его координаты будут многократно изменяться.

        // Классическая ортогонализация: вычитаем проекции на уже построенные
        // направления updatedDirections[0], ..., updatedDirections[j-1].
        for (std::size_t i = 0; i < j; ++i)
        {
            // Здесь проекция берется именно от a[j], а не от уже измененного b.
            // Это соответствует классической схеме Грама-Шмидта, где исходный
            // вектор ортогонализуется относительно уже построенных направлений.
            const double projection = Dot(a[j], updatedDirections[i]);

            // Вычитанием projection * updatedDirections[i] удаляем из b
            // составляющую вдоль уже построенного направления.
            AddScaled(b, -projection, updatedDirections[i]);
        }

        Vector normalized;
        // В этот временный вектор helper-функция запишет уже готовое единичное
        // направление нового базиса.
        if (!OrthogonalizeWithFallback(b,
                                       updatedDirections,
                                       oldDirections,
                                       j,
                                       normalized,
                                       error))
        {
            // Ошибка пробрасывается наверх без изменений, потому что текст уже
            // сформирован в OrthogonalizeWithFallback(...).
            return false;
        }

        // После успешной нормировки записываем очередное направление в новый
        // базис и переходим к следующему j.
        updatedDirections[j] = normalized;
    }

    // На выходе newDirections получает уже полностью собранный и
    // ортонормированный набор новых направлений.
    newDirections = updatedDirections;
    return true;
}
